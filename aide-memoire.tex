\begin{multicols}{2}
  \section*{Indiçage des vecteurs}

  \begin{description}
  \item[\code{x[i]}] élément $i$
  \item[\code{x[c(2, 7)]}] éléments spécifiques
  \item[\code{x[-i]}] tous sauf l'élément $i$
  \item[\code{x[-c(2, 7)]}] tous sauf les éléments spécifiques
  \item[\code{x[\meta{condition}]}] éléments satisfaisant la
    \meta{condition} (vecteur booléen)
  \item[\code{x["nom"]}] élément étiqueté \code{"nom"}
  \end{description}
\end{multicols}
